# Step 3: Display the schema of cars_database.db

import sqlite3
import pandas as pd

# Connect to the SQLite database
conn = sqlite3.connect('cars_database.db')

# Create a cursor object
cursor = conn.cursor()

# Fetch all table names from the database
cursor.execute("SELECT name FROM sqlite_master WHERE type='table';")
tables = cursor.fetchall()

# Prepare a list to store schema details
schema_info = []

# Loop through each table to get its columns and data types
for table in tables:
    table_name = table[0]
    cursor.execute(f"PRAGMA table_info({table_name});")
    columns = cursor.fetchall()
    for col in columns:
        schema_info.append((table_name, col[1], col[2]))  # (Table, Column, Data Type)

# Convert schema info into a pandas DataFrame
dfs = pd.DataFrame(schema_info, columns=['Table', 'Column', 'Data Type'])

# Display the schema DataFrame
dfs

import sqlite3

# ---------------------------------
# CREATE DATABASE
# ---------------------------------
conn = sqlite3.connect("car_database.db")
cursor = conn.cursor()

# ---------------------------------
# CREATE TABLES
# ---------------------------------

cursor.execute("""
CREATE TABLE IF NOT EXISTS Brands (
    brand_id INTEGER PRIMARY KEY,
    brand_name VARCHAR(50) NOT NULL
);
""")

cursor.execute("""
CREATE TABLE IF NOT EXISTS Manufacture_Plant (
    manufacture_plant_id INTEGER PRIMARY KEY,
    company_owned BOOLEAN NOT NULL,
    plant_location VARCHAR(100),
    plant_name VARCHAR(50) NOT NULL,
    plant_type VARCHAR(10) NOT NULL
);
""")

cursor.execute("""
CREATE TABLE IF NOT EXISTS Models (
    model_id INTEGER PRIMARY KEY,
    brand_id INTEGER NOT NULL,
    model_base_price INTEGER NOT NULL,
    model_name VARCHAR(100) NOT NULL,
    FOREIGN KEY (brand_id) REFERENCES Brands(brand_id)
);
""")

cursor.execute("""
CREATE TABLE IF NOT EXISTS Car_Options (
    option_set_id INTEGER PRIMARY KEY,
    chassis_id INTEGER,
    color VARCHAR(50),
    engine_id INTEGER,
    option_set_price INTEGER NOT NULL,
    premium_sound_id INTEGER,
    transmission_id INTEGER
);
""")

cursor.execute("""
CREATE TABLE IF NOT EXISTS Car_Vins (
    vin VARCHAR(17) PRIMARY KEY,
    manufacture_plant_id INTEGER NOT NULL,
    manufactured_date DATE NOT NULL,
    model_id INTEGER NOT NULL,
    option_set_id INTEGER NOT NULL,

    FOREIGN KEY(manufacture_plant_id) REFERENCES Manufacture_Plant(manufacture_plant_id),
    FOREIGN KEY(model_id) REFERENCES Models(model_id),
    FOREIGN KEY(option_set_id) REFERENCES Car_Options(option_set_id)
);
""")

cursor.execute("""
CREATE TABLE IF NOT EXISTS Customers (
    customer_id INTEGER PRIMARY KEY,
    birthdate DATE,
    email VARCHAR(128),
    first_name VARCHAR(50) NOT NULL,
    gender VARCHAR(6),
    household_income INTEGER,
    last_name VARCHAR(50) NOT NULL,
    phone_number VARCHAR(15)
);
""")

cursor.execute("""
CREATE TABLE IF NOT EXISTS Customer_Ownership (
    customer_id INTEGER NOT NULL,
    vin VARCHAR(17) NOT NULL,
    dealer_id INTEGER,
    purchase_date DATE NOT NULL,
    purchase_price INTEGER NOT NULL,
    warranty_expire_date DATE,

    PRIMARY KEY (customer_id, vin),

    FOREIGN KEY(customer_id) REFERENCES Customers(customer_id),
    FOREIGN KEY(vin) REFERENCES Car_Vins(vin)
);
""")

cursor.execute("""
CREATE TABLE IF NOT EXISTS Parts (
    manufacture_plant_id INTEGER,
    manufacture_start_date DATE,
    part_name VARCHAR(100) NOT NULL,
    part_recall BOOLEAN,
    FOREIGN KEY(manufacture_plant_id) REFERENCES Manufacture_Plant(manufacture_plant_id)
);
""")

# ---------------------------------
# SAVE & CLOSE
# ---------------------------------
conn.commit()
conn.close()

print("Database created successfully as car_database.db")

import sqlite3
import pandas as pd
import matplotlib.pyplot as plt

# Connect to DB
conn = sqlite3.connect("car_database.db")

# SQL to compute average purchase price per dealer
query = """
SELECT 
    d.dealer_name,
    AVG(co.purchase_price) AS avg_purchase_price
FROM Customer_Ownership co
JOIN Dealers d 
    ON co.dealer_id = d.dealer_id
GROUP BY d.dealer_name
ORDER BY avg_purchase_price DESC;
"""

df_avg = pd.read_sql_query(query, conn)
conn.close()

# Plot
plt.figure(figsize=(10, 5))
plt.bar(df_avg["dealer_name"], df_avg["avg_purchase_price"])
plt.title("Average Purchase Price for Each Dealer")
plt.xlabel("Dealer")
plt.ylabel("Average Purchase Price ($)")
plt.xticks(rotation=45, ha='right')
plt.tight_layout()
plt.show()
